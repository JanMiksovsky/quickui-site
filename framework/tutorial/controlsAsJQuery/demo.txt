var $button = BasicButton.create()
    .content( "Toggle" )
    .click( function() {
        $comboBox.toggle();
    });

var $comboBox = ColorSwatchComboBox.create()
    .content( "Orange" )
    .items([
        "Blue",
        "Green",
        "Orange",
        "Red"
    ]);

$demo.append(
    $( "<div/>" ).append( $button ),
    $comboBox
);
